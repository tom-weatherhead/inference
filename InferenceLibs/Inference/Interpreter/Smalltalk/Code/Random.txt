(class Rand Object ()
    (seed)
    (define init () (begin (initRand self 1) self))
    (define initRand (n) (set seed n))
    (define nextRand () (set seed (mod (+ (* seed 9) 5) 1024)))
)
(set r (init (new Rand)))
(nextRand r)
(nextRand r)
